int main(){
    int n;
    cin >> n;
    int max = -10000;
    int min = 10000;
    int sum = 0;
    for(int i = 0; i < n; i++){
        int t;
        cin >> t;
        sum += t;
        if(t > max){
            max = t;
        }
        if(t < min){
            min = t;
        }
    }
    cout << max << endl << min << endl << sum << endl;
    return 0;
}
